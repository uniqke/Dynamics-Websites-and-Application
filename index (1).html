let employeesData = []; async function fetchEmployees() { try { const response = await fetch("employees.json"); if (!response.ok) { throw new Error("Failed to fetch employee data"); } const data = await response.json(); const employees = data.employees; processEmployees(employees); } catch (error) { console.error("Error fetching employees:", error); document.getElementById("demo").innerHTML = `
  • Could not load employee data.
    • `; } } function processEmployees(employees) { const sortedEmployees = [...employees].sort((a, b) => a.age - b.age); employeesData = sortedEmployees; displayEmployees(sortedEmployees); showFullNames(sortedEmployees); showTotalSalary(sortedEmployees); showHighEarners(sortedEmployees); saveToLocalStorage(sortedEmployees); } function displayEmployees(employees) { const demo = document.getElementById("demo"); demo.innerHTML = ""; employees.forEach(employee => { const li = document.createElement("li"); li.className = "list-group-item d-flex justify-content-between align-items-center"; li.innerHTML = `
    ${employee.firstName} ${employee.lastName}
    Age: ${employee.age} | Department: ${employee.department}
    $${employee.salary} `; demo.appendChild(li); }); } function showFullNames(employees) { const fullNames = employees.map(({ firstName, lastName }) => `${firstName} ${lastName}`); console.log("Full Names:", fullNames); } function showTotalSalary(employees) { const totalSalary = employees.reduce((total, employee) => total + employee.salary, 0); document.getElementById("totalSalary").textContent = `Total Salary: $${totalSalary}`; } function showHighEarners(employees) { const highEarners = employees.filter(employee => employee.salary > 55000); const names = highEarners.map(employee => `${employee.firstName} ${employee.lastName}`); console.log("Employees earning more than $55,000:", names); } function saveToLocalStorage(employees) { localStorage.setItem("sortedEmployees", JSON.stringify(employees)); } document.getElementById("showStorage").addEventListener("click", () => { const storedData = JSON.parse(localStorage.getItem("sortedEmployees")); console.log("Stored Data:", storedData); }); document.getElementById("removeStorage").addEventListener("click", () => { localStorage.removeItem("sortedEmployees"); console.log("sortedEmployees removed"); }); document.getElementById("clearStorage").addEventListener("click", () => { localStorage.clear(); console.log("All localStorage cleared"); }); document.getElementById("search").addEventListener("input", function () { const searchValue = this.value.toLowerCase(); const filteredEmployees = employeesData.filter(employee => `${employee.firstName} ${employee.lastName}`.toLowerCase().includes(searchValue) ); displayEmployees(filteredEmployees); }); fetchEmployees();
